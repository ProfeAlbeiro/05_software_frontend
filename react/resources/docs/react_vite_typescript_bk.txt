              import axios from 'axios';

              const BASE_URL = import.meta.env.VITE_API_URL || 'http://localhost:3000/api';

              export const apiDelivery = axios.create({
                  baseURL: BASE_URL,
                  headers: {
                      'Content-Type': 'application/json',
                  },
                  timeout: 10000,
              });

              // Interceptor para agregar token a las requests
              apiDelivery.interceptors.request.use(
                  (config) => {
                      const token = localStorage.getItem('authToken');
                      if (token) {
                          config.headers.Authorization = `Bearer ${token}`;
                      }
                      return config;
                  },
                  (error) => {
                      return Promise.reject(error);
                  }
              );

              // Interceptor para manejar respuestas
              apiDelivery.interceptors.response.use(
                  (response) => response,
                  (error) => {
                      if (error.response?.status === 401) {
                          localStorage.removeItem('authToken');
                          window.location.href = '/login';
                      }
                      return Promise.reject(error);
                  }
              );

              export default apiDelivery;

              import { AuthRepository } from '@domain/repositories/AuthRepository';
              import { User, AuthResponse, LoginRequest, RegisterRequest } from '@domain/entities/User';
              import { apiDelivery } from '@data/sources/remote/api/ApiDelivery';
              import { AuthResponseApi } from '@data/sources/remote/models/ResponseApiDelivery';

              export class AuthRepositoryImpl implements AuthRepository {
                  async login(credentials: LoginRequest): Promise<AuthResponse> {
                      try {
                          const response = await apiDelivery.post<AuthResponseApi>('/auth/login', credentials);
                          
                          if (response.data.success && response.data.token && response.data.user) {
                              localStorage.setItem('authToken', response.data.token);
                              
                              return {
                                  success: true,
                                  message: response.data.message,
                                  user: {
                                      id: response.data.user.id,
                                      name: response.data.user.name,
                                      email: response.data.user.email
                                  },
                                  token: response.data.token
                              };
                          }
                          
                          return {
                              success: false,
                              message: response.data.message || 'Error en el login'
                          };
                      } catch (error: unknown) {
                          const errorMessage = error instanceof Error ? error.message : 'Error de conexiÃ³n';
                          return {
                              success: false,
                              message: errorMessage
                          };
                      }
                  }

                  async register(userData: RegisterRequest): Promise<AuthResponse> {
                      try {
                          const response = await apiDelivery.post<AuthResponseApi>('/auth/register', userData);
                          
                          if (response.data.success && response.data.token && response.data.user) {
                              localStorage.setItem('authToken', response.data.token);
                              
                              return {
                                  success: true,
                                  message: response.data.message,
                                  user: {
                                      id: response.data.user.id,
                                      name: response.data.user.name,
                                      email: response.data.user.email
                                  },
                                  token: response.data.token
                              };
                          }
                          
                          return {
                              success: false,
                              message: response.data.message || 'Error en el registro'
                          };
                      } catch (error: unknown) {
                          const errorMessage = error instanceof Error ? error.message : 'Error de conexiÃ³n';
                          return {
                              success: false,
                              message: errorMessage
                          };
                      }
                  }

                  async logout(): Promise<void> {
                      localStorage.removeItem('authToken');
                      try {
                          await apiDelivery.post('/auth/logout');
                      } catch {
                          // Ignorar errores en logout
                      }
                  }

                  async getCurrentUser(): Promise<User | null> {
                      try {
                          const response = await apiDelivery.get<AuthResponseApi>('/auth/me');
                          return response.data.user ? {
                              id: response.data.user.id,
                              name: response.data.user.name,
                              email: response.data.user.email
                          } : null;
                      } catch {
                          return null;
                      }
                  }

                  async validateToken(token: string): Promise<boolean> {
                      try {
                          const response = await apiDelivery.post<{ valid: boolean }>('/auth/validate', { token });
                          return response.data.valid;
                      } catch {
                          return false;
                      }
                  }
              }
